# 自動把外部導覽屬性填充

可以做一個**輔助方法**，把外部系統的導覽屬性自動載入，示範使用你的 `TaskOrder` 與 `Employee`（跨 `CoreContext`）：

```csharp
public static class ExternalLoader
{
    public static void LoadExternalNavigation(this TaskOrder task, CoreContext coreContext)
    {
        if (task.CreatedBy == null)
            task.CreatedBy = coreContext.Employees
                                .FirstOrDefault(e => e.EmployeeId == task.CreatedById);

        if (task.RequestingEmployee == null)
            task.RequestingEmployee = coreContext.Employees
                                        .FirstOrDefault(e => e.EmployeeId == task.RequestingEmployeeId);

        if (task.Engineers != null && !task.Engineers.Any())
        {
            // 假設你有 TaskOrderEngineers 的關聯表
            var engineerIds = coreContext.TaskOrderEngineers
                                         .Where(te => te.TaskOrderId == task.WorkOrderNo)
                                         .Select(te => te.EmployeeId)
                                         .ToList();

            task.Engineers = coreContext.Engineers
                                 .Where(e => engineerIds.Contains(e.EmployeeId))
                                 .ToList();
        }
    }

    public static void LoadExternalNavigation(this IEnumerable<TaskOrder> tasks, CoreContext coreContext)
    {
        foreach (var task in tasks)
            task.LoadExternalNavigation(coreContext);
    }
}
```

## 使用方式

```csharp
// 先查 MaintiFlowContext
var tasks = maintiFlowContext.TaskOrders.ToList();

// 再載入外部 CoreContext
tasks.LoadExternalNavigation(coreContext);

// 現在 task.CreatedBy、task.RequestingEmployee、task.Engineers 都有值
```

## 特點

- `[NotMapped]` 導覽屬性不會被 EF 嘗試建立 FK。
- Lazy Load 失效，手動載入即可。
- 可批量載入，避免 N+1 查詢問題。

這個方法也可以延伸，用 **Include + Select** 或 **Join** 一次查完外部表，再填入導覽屬性，效率更高。

# 批量一次查完多筆外部導覽屬性的最佳做法，比逐筆查詢快很多

這裡提供一個批量載入外部導覽屬性的範例，針對多筆 `TaskOrder` 一次查完 `CoreContext` 的資料，避免 N+1 問題：

```csharp
public static class ExternalLoader
{
    public static void LoadExternalNavigationBatch(this List<TaskOrder> tasks, CoreContext coreContext)
    {
        var createdByIds = tasks.Select(t => t.CreatedById).Distinct().ToList();
        var requestingIds = tasks.Select(t => t.RequestingEmployeeId).Distinct().ToList();

        // 一次抓所有相關 Employee
        var employees = coreContext.Employees
            .Where(e => createdByIds.Contains(e.EmployeeId) || requestingIds.Contains(e.EmployeeId))
            .ToDictionary(e => e.EmployeeId);

        foreach (var task in tasks)
        {
            if (employees.TryGetValue(task.CreatedById, out var createdBy))
                task.CreatedBy = createdBy;

            if (employees.TryGetValue(task.RequestingEmployeeId, out var requesting))
                task.RequestingEmployee = requesting;
        }

        // Engineers 多對多
        var taskOrderIds = tasks.Select(t => t.WorkOrderNo).ToList();
        var taskEngineers = coreContext.TaskOrderEngineers
            .Where(te => taskOrderIds.Contains(te.TaskOrderId))
            .ToList();

        var engineerIds = taskEngineers.Select(te => te.EmployeeId).Distinct().ToList();
        var engineers = coreContext.Engineers
            .Where(e => engineerIds.Contains(e.EmployeeId))
            .ToDictionary(e => e.EmployeeId);

        var taskDict = tasks.ToDictionary(t => t.WorkOrderNo);
        foreach (var te in taskEngineers)
        {
            if (taskDict.TryGetValue(te.TaskOrderId, out var task) && engineers.TryGetValue(te.EmployeeId, out var eng))
            {
                if (task.Engineers == null)
                    task.Engineers = new List<Engineer>();
                task.Engineers.Add(eng);
            }
        }
    }
}
```

## 使用方式

```csharp
var tasks = maintiFlowContext.TaskOrders.ToList();
tasks.LoadExternalNavigationBatch(coreContext);

// 現在 CreatedBy、RequestingEmployee、Engineers 都一次載入完成
```

## 優點

- **避免逐筆查詢**：一次查完所有相關 Employee 與 Engineer。
- **可擴充**：其他外部導覽屬性也能用同樣邏輯批量載入。
- **控制 `[NotMapped]` 導覽屬性**：不影響 EF 自己的 DB 映射。

# 通用批量外部導覽載入工具，只要傳入 `TaskOrder` 和需要的屬性，就能自動處理所有外部 FK

可以，我幫你設計一個通用的批量外部導航屬性載入工具，核心思路是：

1. 傳入一組實體（如 `List<TaskOrder>`）
2. 指定外部導航屬性與對應 FK
3. 自動從外部 `DbContext` 批量抓資料，一次載入完成

範例如下：

```csharp
public static class ExternalNavigationLoader
{
    public static void LoadExternalNavigation<T, TKey, TNav>(
        this List<T> entities,
        DbContext externalContext,
        Func<T, TKey> fkSelector,
        Action<T, TNav> navSetter,
        DbSet<TNav> navDbSet,
        Func<TNav, TKey> navKeySelector
    ) where T : class
      where TNav : class
    {
        // 1. 取得所有 FK 值
        var fkValues = entities.Select(fkSelector).Distinct().ToList();

        // 2. 批量查詢外部資料
        var navData = navDbSet
            .Where(n => fkValues.Contains(navKeySelector(n)))
            .ToDictionary(navKeySelector);

        // 3. 一次性設定導航屬性
        foreach (var entity in entities)
        {
            var fk = fkSelector(entity);
            if (navData.TryGetValue(fk, out var nav))
                navSetter(entity, nav);
        }
    }

    // 專用於多對多關聯
    public static void LoadExternalManyToMany<T, TNav, TLink, TKey>(
        this List<T> entities,
        DbContext externalContext,
        Func<T, TKey> entityKeySelector,
        Func<TLink, TKey> linkEntityKeySelector,
        Func<TLink, string> linkNavKeySelector,
        DbSet<TLink> linkDbSet,
        DbSet<TNav> navDbSet,
        Action<T, ICollection<TNav>> navSetter,
        Func<TNav, string> navKeySelector
    ) where T : class
      where TNav : class
      where TLink : class
    {
        var entityKeys = entities.Select(entityKeySelector).Distinct().ToList();
        var links = linkDbSet.Where(l => entityKeys.Contains(linkEntityKeySelector(l))).ToList();
        var navIds = links.Select(linkNavKeySelector).Distinct().ToList();
        var navData = navDbSet.Where(n => navIds.Contains(navKeySelector(n)))
                              .ToDictionary(navKeySelector);

        var entityDict = entities.ToDictionary(entityKeySelector);
        foreach (var link in links)
        {
            var entityKey = linkEntityKeySelector(link);
            var navKey = linkNavKeySelector(link);

            if (entityDict.TryGetValue(entityKey, out var entity) && navData.TryGetValue(navKey, out var nav))
            {
                var coll = navSetterGet(entity, navSetter);
                coll.Add(nav);
            }
        }

        // 取得或初始化集合
        static ICollection<TNav> navSetterGet(T entity, Action<T, ICollection<TNav>> setter)
        {
            var coll = new List<TNav>();
            setter(entity, coll);
            return coll;
        }
    }
}
```

## 使用範例

```csharp

    coreContext,
```

這套工具的好處：

- 完全 **批量載入**，不會觸發 N+1
- 支援 **單對單 / 一對多** 以及 **多對多**
- 導航屬性可以 `[NotMapped]`，不必依賴 EF Lazy Load

# 更簡潔的泛型版本，只要一行就能自動判斷 FK 與導航屬性，自動載入

可以，我幫你改成進階泛型版，目標是：

- 傳入實體集合與外部 `DbContext`
- 自動找出 FK 與導航屬性（透過屬性名稱或委派指定）
- 一行呼叫就能載入單對單 / 一對多 / 多對多
- 支援 `[NotMapped]` 導航屬性

範例如下：

```csharp
public static class ExternalNavigationLoader
{
    // 單對單 / 一對多
    public static void LoadNavigation<T, TNav, TKey>(
        this IEnumerable<T> entities,
        DbContext externalContext,
        Func<T, TKey> fkSelector,
        Expression<Func<T, TNav>> navProperty,
        DbSet<TNav> navDbSet,
        Func<TNav, TKey> navKeySelector
    ) where T : class
      where TNav : class
    {
        var entityList = entities.ToList();
        var fkValues = entityList.Select(fkSelector).Distinct().ToList();
        var navData = navDbSet.Where(n => fkValues.Contains(navKeySelector(n)))
                              .ToDictionary(navKeySelector);

        var prop = (PropertyInfo)((MemberExpression)navProperty.Body).Member;
        foreach (var entity in entityList)
        {
            var fk = fkSelector(entity);
            if (navData.TryGetValue(fk, out var nav))
                prop.SetValue(entity, nav);
        }
    }

    // 多對多
    public static void LoadManyToMany<T, TNav, TLink, TKey>(
        this IEnumerable<T> entities,
        DbContext externalContext,
        Func<T, TKey> entityKeySelector,
        DbSet<TLink> linkDbSet,
        Func<TLink, TKey> linkEntityKeySelector,
        Func<TLink, string> linkNavKeySelector,
        DbSet<TNav> navDbSet,
        Func<TNav, string> navKeySelector,
        Expression<Func<T, ICollection<TNav>>> navProperty
    ) where T : class
      where TNav : class
      where TLink : class
    {
        var entityList = entities.ToList();
        var entityKeys = entityList.Select(entityKeySelector).Distinct().ToList();
        var links = linkDbSet.Where(l => entityKeys.Contains(linkEntityKeySelector(l))).ToList();
        var navIds = links.Select(linkNavKeySelector).Distinct().ToList();
        var navData = navDbSet.Where(n => navIds.Contains(navKeySelector(n)))
                              .ToDictionary(navKeySelector);

        var prop = (PropertyInfo)((MemberExpression)navProperty.Body).Member;
        var entityDict = entityList.ToDictionary(entityKeySelector);

        foreach (var link in links)
        {
            var entityKey = linkEntityKeySelector(link);
            var navKey = linkNavKeySelector(link);
            if (entityDict.TryGetValue(entityKey, out var entity) && navData.TryGetValue(navKey, out var nav))
            {
                var coll = (ICollection<TNav>)prop.GetValue(entity);
                if (coll == null)
                {
                    coll = new List<TNav>();
                    prop.SetValue(entity, coll);
                }
                coll.Add(nav);
            }
        }
    }
}
```

# 自動 FK 與導航屬性推斷版，連 `Func` 都不用寫，完全一行就搞定

那我幫你設計一個 **完全自動版**，目標：

- 自動找出 `TaskOrder` 的 FK 與導航屬性
- 自動匹配外部 `DbContext` 的 `DbSet`
- 支援 `[NotMapped]` 導航屬性
- 一行呼叫即可載入全部外部關聯

```csharp
public static class ExternalNavigationLoaderAuto
{
    public static void LoadAllExternal<T>(this IEnumerable<T> entities, DbContext externalContext) where T : class
    {
        var entityList = entities.ToList();
        var entityType = typeof(T);

        // 找出所有虛擬導航屬性
        var navProps = entityType.GetProperties()
            .Where(p => p.PropertyType.IsClass && p.GetGetMethod().IsVirtual && p.GetCustomAttribute<NotMappedAttribute>() != null);

        foreach (var navProp in navProps)
        {
            // 嘗試找到 FK 屬性
            string fkName = navProp.Name + "Id";
            var fkProp = entityType.GetProperty(fkName);
            if (fkProp == null) continue;

            // 對應外部 DbSet
            var navType = navProp.PropertyType;
            var dbSetProp = externalContext.GetType()
                .GetProperties()
                .FirstOrDefault(p => p.PropertyType.IsGenericType
                                     && p.PropertyType.GetGenericArguments()[0] == navType);

            if (dbSetProp == null) continue;

            dynamic dbSet = dbSetProp.GetValue(externalContext);

            // 批量載入
            var fkValues = entityList.Select(e => fkProp.GetValue(e)).Distinct().ToList();
            var navData = dbSet.Where(n => fkValues.Contains(n.Id)).ToDictionary(n => n.Id);

            foreach (var entity in entityList)
            {
                var fkValue = fkProp.GetValue(entity);
                if (navData.TryGetValue(fkValue, out var nav))
                    navProp.SetValue(entity, nav);
            }
        }
    }
}
```

## 使用方式

```csharp
var tasks = maintiFlowContext.TaskOrders.ToList();
tasks.LoadAllExternal(coreContext);
```

這樣：

- 所有 `[NotMapped]` + `virtual` 的外部導航屬性都會自動載入
- 不需要手動指定 FK、DbSet 或委派
- 適合你的多 `DbContext` 情境

# 多對多自動載入版本，這樣連 `Engineers` 也可以一行自動載入

```csharp
public static class ExternalNavigationLoaderAuto
{
    public static void LoadAllExternal<T>(this IEnumerable<T> entities, DbContext externalContext) where T : class

        foreach (var navProp in navProps)
        {

                    .GetProperties()

                {
                    // 此處假設 join table 的關聯先手動建立，EF6 的多對多關聯會自動填充
                    // 如果 join table 也在外部 DB，可能需額外實作

}
```

# 完全自動的多對多 join table 批量載入，讓 `Engineers` 也一行就載入完成

好的，我來幫你做一個 **完全自動多對多批量載入** 的版本，連 `TaskOrder.Engineers` 這種多對多也能一次搞定。思路是：

1. 偵測 `[NotMapped]` + `virtual` 屬性
2. 判斷屬性是單對一、一對多或多對多
3. 多對多自動找 join table，批量查出對應關聯
4. 填回原實體

範例程式如下：

```csharp
public static class ExternalNavigationLoaderAuto
{
    public static void LoadAllExternal<T>(this IEnumerable<T> entities, DbContext externalContext) where T : class
    {
        var entityList = entities.ToList();
        var entityType = typeof(T);

        // 找出所有虛擬導航屬性
        var navProps = entityType.GetProperties()
            .Where(p => p.GetGetMethod().IsVirtual && p.GetCustomAttribute<NotMappedAttribute>() != null);

        foreach (var navProp in navProps)
        {
            var propType = navProp.PropertyType;

            // 多對多 ICollection<T>?
            if (propType.IsGenericType && typeof(IEnumerable<>).IsAssignableFrom(propType.GetGenericTypeDefinition()))
            {
                var itemType = propType.GetGenericArguments()[0];

                // 嘗試找到對應外部 DbSet
                var dbSetProp = externalContext.GetType()
                    .GetProperties()
                    .FirstOrDefault(p => p.PropertyType.IsGenericType
                                         && p.PropertyType.GetGenericArguments()[0] == itemType);
                if (dbSetProp == null) continue;

                dynamic dbSet = dbSetProp.GetValue(externalContext);

                // 假設 join table 名稱規則為 "EntityNameItemTypeName" (如 TaskOrderEngineer)
                string joinTableName = entityType.Name + itemType.Name;
                var joinTable = externalContext.Database.SqlQuery<dynamic>($"SELECT * FROM {joinTableName}").ToList();

                // 建立字典快速查找
                var map = joinTable.GroupBy(j => (int)j[entityType.Name + "Id"])
                                   .ToDictionary(g => g.Key, g => g.Select(x => (string)x[itemType.Name + "Id"]).ToList());

                foreach (var entity in entityList)
                {
                    var entityId = entityType.GetProperty("Id")?.GetValue(entity)
                                  ?? entityType.GetProperty(entityType.Name + "Id")?.GetValue(entity);

                    if (entityId != null && map.TryGetValue((int)entityId, out var ids))
                    {
                        var collection = (IList<object>)Activator.CreateInstance(typeof(List<>).MakeGenericType(itemType));
                        foreach (var id in ids)
                        {
                            var nav = dbSet.SingleOrDefault(d => d.EmployeeId == id || d.Id == id);
                            if (nav != null)
                                collection.Add(nav);
                        }
                        navProp.SetValue(entity, collection);
                    }
                }

                continue;
            }

            // 單對一或一對多
            string fkName = navProp.Name + "Id";
            var fkProp = entityType.GetProperty(fkName);
            if (fkProp == null) continue;

            // 對應外部 DbSet
            var navType = navProp.PropertyType;
            var dbSetSingle = externalContext.GetType()
                .GetProperties()
                .FirstOrDefault(p => p.PropertyType.IsGenericType
                                     && p.PropertyType.GetGenericArguments()[0] == navType);
            if (dbSetSingle == null) continue;

            dynamic dbSetObj = dbSetSingle.GetValue(externalContext);

            // 批量載入
            var fkValues = entityList.Select(e => fkProp.GetValue(e)).Distinct().ToList();
            var navData = dbSetObj.Where(n => fkValues.Contains(n.Id)).ToDictionary(n => n.Id);

            foreach (var entity in entityList)
            {
                var fkValue = fkProp.GetValue(entity);
                if (navData.TryGetValue(fkValue, out var nav))
                    navProp.SetValue(entity, nav);
            }
        }
    }
}
```


# 支援不同外部 join table 名稱規則，不用固定 TaskOrderEngineer 這種命名

好的，我來幫你把 **多對多 join table 名稱可自訂化** 加進去。思路是：

1. 在呼叫 `LoadAllExternal` 時可以提供一個字典，key 是「實體名稱 + 導航屬性名稱」，value 是對應的 join table 名稱。
2. 程式會先查字典，找不到才用預設規則。
3. 其餘流程不變，仍自動批量載入單對一、一對多與多對多。

修改後程式範例如下：

```csharp
public static class ExternalNavigationLoaderAuto
{
    public static void LoadAllExternal<T>(
        this IEnumerable<T> entities,
        DbContext externalContext,
        Dictionary<string, string> joinTableMap = null) where T : class
    {
        var entityList = entities.ToList();
        var entityType = typeof(T);

        var navProps = entityType.GetProperties()
            .Where(p => p.GetGetMethod().IsVirtual && p.GetCustomAttribute<NotMappedAttribute>() != null);

        foreach (var navProp in navProps)
        {
            var propType = navProp.PropertyType;

            // 多對多 ICollection<T>?
            if (propType.IsGenericType && typeof(IEnumerable<>).IsAssignableFrom(propType.GetGenericTypeDefinition()))
            {
                var itemType = propType.GetGenericArguments()[0];
                var dbSetProp = externalContext.GetType()
                    .GetProperties()
                    .FirstOrDefault(p => p.PropertyType.IsGenericType
                                         && p.PropertyType.GetGenericArguments()[0] == itemType);
                if (dbSetProp == null) continue;

                dynamic dbSet = dbSetProp.GetValue(externalContext);

                // 取得 join table 名稱，先看字典
                string key = entityType.Name + navProp.Name;
                string joinTableName = joinTableMap != null && joinTableMap.TryGetValue(key, out var jtName)
                    ? jtName
```

# 支援外部表的自動 FK 推斷，讓單對一、一對多也不用寫 FK 名稱就能對應

好的，我們可以再把 **自動推斷 FK** 加進去。思路是：

1. 對於每個 `[NotMapped]` 的導航屬性，如果沒有顯式提供 FK 名稱，程式會自動去找同名屬性加 `Id` 或 `名稱+Id` 的欄位。
2. 多對多的 join table 名稱還是先用字典，如果找不到才用預設規則。
3. 單對一、一對多、以及多對多都自動匹配，不需要手動指定 FK 屬性。

修改範例核心程式如下：

```csharp
foreach (var navProp in navProps)
{
    var propType = navProp.PropertyType;

    // 多對多 ICollection<T>
    if (propType.IsGenericType && typeof(IEnumerable<>).IsAssignableFrom(propType.GetGenericTypeDefinition()))
    {
        var itemType = propType.GetGenericArguments()[0];
        var dbSetProp = externalContext.GetType()
            .GetProperties()
            .FirstOrDefault(p => p.PropertyType.IsGenericType
                                 && p.PropertyType.GetGenericArguments()[0] == itemType);
        if (dbSetProp == null) continue;
        dynamic dbSet = dbSetProp.GetValue(externalContext);

        string key = entityType.Name + navProp.Name;
        string joinTableName = joinTableMap != null && joinTableMap.TryGetValue(key, out var jtName)
            ? jtName
            : entityType.Name + itemType.Name;

        var joinTable = externalContext.Database.SqlQuery<dynamic>($"SELECT * FROM {joinTableName}").ToList();
        string leftKey = entityType.Name + "Id";
        string rightKey = itemType.Name + "Id";

        var map = joinTable.GroupBy(j => (int)j[leftKey])
                           .ToDictionary(g => g.Key, g => g.Select(x => (string)x[rightKey]).ToList());

        foreach (var entity in entityList)
        {
            var entityId = entityType.GetProperty("Id")?.GetValue(entity)
                          ?? entityType.GetProperty(leftKey)?.GetValue(entity);
            if (entityId != null && map.TryGetValue((int)entityId, out var ids))
            {
                var collection = (IList<object>)Activator.CreateInstance(typeof(List<>).MakeGenericType(itemType));
                foreach (var id in ids)
                {
                    var nav = dbSet.SingleOrDefault(d => d.EmployeeId == id || d.Id == id);
                    if (nav != null)
                        collection.Add(nav);
                }
                navProp.SetValue(entity, collection);
            }
        }
        continue;
    }

    // 單對一 / 一對多
    // 自動找 FK
    string fkName = navProp.Name + "Id";
    var fkProp = entityType.GetProperty(fkName)
                 ?? entityType.GetProperties().FirstOrDefault(p => p.Name.EndsWith("Id") && p.Name.StartsWith(navProp.Name));
    if (fkProp == null) continue;

    var navType = navProp.PropertyType;
    var dbSetSingle = externalContext.GetType()
        .GetProperties()
        .FirstOrDefault(p => p.PropertyType.IsGenericType
                             && p.PropertyType.GetGenericArguments()[0] == navType);
    if (dbSetSingle == null) continue;

    dynamic dbSetObj = dbSetSingle.GetValue(externalContext);
    var fkValues = entityList.Select(e => fkProp.GetValue(e)).Distinct().ToList();
    var navData = dbSetObj.Where(n => fkValues.Contains(n.Id) || fkValues.Contains(n.EmployeeId))
                           .ToDictionary(n => n.Id != null ? n.Id : n.EmployeeId);

    foreach (var entity in entityList)
    {
        var fkValue = fkProp.GetValue(entity);
        if (navData.TryGetValue(fkValue, out var nav))
            navProp.SetValue(entity, nav);
    }
}
```

# 自訂 join table 名稱優缺點分析

## 不固定 join table 名稱的好處

1. **彈性高**
    - 不需要預先在程式或資料庫中定義 join table 的名稱，隨時可以對應多對多關聯。
    - 對多個專案或不同資料庫結構特別方便，尤其外部資料庫結構你無法改動時。
2. **減少耦合**
    - 程式不依賴特定 join table 名稱，改動多對多關聯只要更新屬性或 FK，資料庫結構不影響程式邏輯。
3. **方便快速原型**
    - 在開發早期或測試階段，不用思考 join table 命名規則，快速建立多對多關聯。
4. **外部資料庫導入友好**
    - 如果多對多的關聯表名稱不是程式自訂，而是外部已有資料表，也能透過自動偵測或設定映射來使用。

## 缺點

1. **可讀性差**
    - 自動生成的名稱通常是 EF 內部規則，例如 `TaskOrder_Engineer` 或 `EngineerTaskOrders`，不直觀，資料庫開發或報表人員可能難以理解。
2. **版本控制不穩定**
    - EF Migration 在不同環境或修改模型後，可能生成不同 join table 名稱，導致 Migration 無法正確比對或升級資料庫。
3. **難以維護 SQL**
    - 若要寫原生 SQL 查詢 join table，自動命名可能和你想像不同，需要先查 EF 生成的名稱。
4. **跨 DbContext 或外部資料庫問題**
    - 自動命名 join table 可能與另一個 DbContext 或外部資料庫已有表重名，容易產生衝突。
5. **限制自訂欄位**
    - 如果將來需要在 join table 上加額外欄位（例如關聯建立時間、狀態），自動生成的 join table 不方便直接擴充，需要改成顯式實體。

## 什麼時候一定要自訂 join table 名稱

1. **資料庫規範要求**
    - 公司或專案有固定命名規則，需要 join table 名稱符合規範。
2. **多個 join table 可能同名**
    - 如果不同多對多關聯可能被 EF 自動命名成相同名字（尤其在多個 DbContext 或 schema 中），必須自訂避免衝突。
3. **跨資料庫或報表需求**
    - 需要明確 join table 名稱給報表、查詢或其他系統使用，方便 SQL 調試。
4. **遷移與版本控制**
    - EF Migration 生成 join table 時，如果沒有固定名稱，下一次 Migration 可能生成不同名字，容易導致資料庫版本不一致。

簡單來說，**不固定命名的好處是彈性、快速、方便外部資料導入**；**自訂命名的場景多是標準化、避免衝突或報表需求**。

總結：**彈性大、開發快，但維護和可控性差**。如果專案會長期維護、多人協作，通常建議自訂 join table 名稱。

```csharp
public static class ExternalDbHelper
{
    public static void SaveExternalEntity<TExternal>(
        TExternal entity,
        DbContext externalContext
    ) where TExternal : class
    {
        if (entity == null) return;

        if (entry.State == EntityState.Detached)
            externalContext.Set<TExternal>().Add(entity);
        else

        externalContext.SaveChanges();
    }

    public static void SaveExternalEntities<TExternal>(
        IEnumerable<TExternal> entities,
        DbContext externalContext
    ) where TExternal : class
    {

    }
```

# 自動辨識主鍵型別（int、string…），完全不用手動寫 `Id` 或 `EmployeeId`。這樣就完全泛型化

這裡我幫你改成完全泛型化、自動辨識主鍵型別（int、string…）的版本。核心改進：

- 不再硬編寫 `Id` 或 `EmployeeId`，自動抓主鍵屬性 `[Key]`
- 支援單對一、多對多、多對一
- 外部資料庫任意主鍵型別（int、string、Guid…）

```csharp
public static class ExternalNavigationLoader
{
    public static void LoadAllExternal<T>(
    this IEnumerable<T> entities,
    DbContext externalContext,
    Dictionary<string, string> joinTableMap = null) where T : class
    {
        var entityList = entities.ToList();
        if (!entityList.Any()) return;

        var entityType = typeof(T);

        var navProps = entityType.GetProperties()
            .Where(p => p.GetGetMethod().IsVirtual && p.GetCustomAttribute<NotMappedAttribute>() != null)
            .ToList();
        if (!navProps.Any()) return;

        // 找主鍵屬性
        var entityKeyProp = entityType.GetProperties().FirstOrDefault(p => p.GetCustomAttribute<KeyAttribute>() != null);

        foreach (var navProp in navProps)
        {
            var propType = navProp.PropertyType;

            // 多對多 ICollection<T>
            if (propType.IsGenericType && typeof(IEnumerable<>).IsAssignableFrom(propType.GetGenericTypeDefinition()))
            {
                var itemType = propType.GetGenericArguments()[0];
                var dbSetProp = externalContext.GetType().GetProperties()
                    .FirstOrDefault(p => p.PropertyType.IsGenericType && p.PropertyType.GetGenericArguments()[0] == itemType);
                if (dbSetProp == null) continue;

                dynamic dbSet = dbSetProp.GetValue(externalContext);

                string key = entityType.Name + navProp.Name;
                string joinTableName = joinTableMap != null && joinTableMap.TryGetValue(key, out var jtName)
                    ? jtName
                    : entityType.Name + itemType.Name;

                var joinTable = externalContext.Database.SqlQuery<dynamic>($"SELECT * FROM {joinTableName}").ToList();

                var map = joinTable.GroupBy(j => j[entityKeyProp.Name])
                                    .ToDictionary(g => g.Key, g => g.Select(x => x[itemType.Name + "Id"]).ToList());

                foreach (var entity in entityList)
                {
                    var entityId = entityKeyProp.GetValue(entity);
                    if (entityId != null && map.TryGetValue(entityId, out var ids))
                    {
                        var collection = (IList)Activator.CreateInstance(typeof(List<>).MakeGenericType(itemType));
                        foreach (var id in ids)
                        {
                            object nav = null;
                            foreach (var d in dbSet) // dbSet 是 dynamic DbSet
                            {
                                PropertyInfo pkProp = null;
                                var props = d.GetType().GetProperties();
                                for (int i = 0; i < props.Length; i++)
                                {
                                    var attr = props[i].GetCustomAttributes(typeof(KeyAttribute), true);
                                    if (attr != null && attr.Length > 0)
                                    {
                                        pkProp = props[i];
                                        break;
                                    }
                                }
                                if (pkProp == null) continue;

                                var pkValue = pkProp.GetValue(d);
                                if (pkValue != null && pkValue.Equals(id))
                                {
                                    nav = d;
                                    break;
                                }
                            }
                            if (nav != null) collection.Add(nav);
                        }
                        navProp.SetValue(entity, collection);
                    }
                }

                continue;
            }

            // 單對一或多對一
            string fkName = navProp.Name + "Id";
            var fkProp = entityType.GetProperty(fkName);
            if (fkProp == null) continue;

            var navType = navProp.PropertyType;
            var dbSetSingleProp = externalContext.GetType().GetProperties()
                .FirstOrDefault(p => p.PropertyType.IsGenericType && p.PropertyType.GetGenericArguments()[0] == navType);
            if (dbSetSingleProp == null) continue;

            dynamic dbSetObj = dbSetSingleProp.GetValue(externalContext);

            var fkValues = entityList.Select(e => fkProp.GetValue(e)).Distinct().ToList();

            // 找導航屬性主鍵
            var navKeyProp = navType.GetProperties().FirstOrDefault(p => p.GetCustomAttribute<KeyAttribute>() != null);
            var navData = new Dictionary<object, object>();
            foreach (var d in (IEnumerable)dbSetObj)
            {
                var key = navKeyProp.GetValue(d);
                if (key != null && !navData.ContainsKey(key))
                    navData.Add(key, d);
            }

            foreach (var entity in entityList)
            {
                var fkValue = fkProp.GetValue(entity);
                if (fkValue != null && navData.TryGetValue(fkValue, out var nav))
                    navProp.SetValue(entity, nav);
            }
        }
    }
}
```

這個版本完全不用管主鍵型別是 `int`、`string` 還是其他，只要 `[Key]` 標記正確，就能自動對應。
